Mehrere grep-basierte Konsistenz-Checks:

```bash
SK=/Users/ap/.claude/skills/archicad

# 1. SKILL.md verweist auf alle 4 Reference-Files
for f in mcp-conventions workflow-context bulk-operations self-improvement; do
  grep -q "reference/$f.md" "$SK/SKILL.md" || echo "FAIL: SKILL.md verweist nicht auf $f"
done

# 2. SKILL.md enthält alle 5 Safety-Regel-Anker
for s in SAFE-01 SAFE-02 SAFE-03 SAFE-04 SAFE-05; do
  grep -q "$s" "$SK/SKILL.md" || echo "FAIL: SAFE-Anker $s fehlt"
done

# 3. Keine nested Reference-Links (one-level-deep)
grep -E "reference/[a-z-]+/[a-z-]+\.md" "$SK"/*.md "$SK"/reference/*.md && echo "FAIL: nested reference link gefunden"

# 4. Workflow-context.md hat TODOs (D-08)
TODO_COUNT=$(grep -c "TODO" "$SK/reference/workflow-context.md")
[ "$TODO_COUNT" -ge 6 ] || echo "FAIL: workflow-context.md hat zu wenige TODO-Marker ($TODO_COUNT, erwartet ≥6)"

# 5. Self-improvement.md hat KEINE TODOs (Methodik ist fertig)
SELF_TODOS=$(grep -c "TODO" "$SK/reference/self-improvement.md")
[ "$SELF_TODOS" -eq 0 ] || echo "FAIL: self-improvement.md hat $SELF_TODOS TODOs (erwartet 0)"

# 6. Alle MCP-Toolnamen voll-qualifiziert
grep -E "(^|[^_])archicad_(call_tool|discover_tools|list_active_archicads)" "$SK"/*.md "$SK"/reference/*.md && echo "FAIL: unqualifizierter MCP-Toolname gefunden"

# 7. TOC in jedem Reference-File >100 Zeilen
for f in mcp-conventions workflow-context bulk-operations self-improvement; do
  LC=$(wc -l < "$SK/reference/$f.md")
  if [ "$LC" -gt 100 ]; then
    head -20 "$SK/reference/$f.md" | grep -q "Inhaltsverzeichnis" || echo "FAIL: $f.md hat $LC Zeilen aber kein TOC"
  fi
done

echo "Checks abgeschlossen."
```
